63/** \brief convert a double to a string, using the loacle "C"
64 * \ingroup jkqtptools_string
65 */
66inline QString JKQTPCDoubleToQString(double value) {
67 QLocale loc=QLocale::c();
68 loc.setNumberOptions(QLocale::OmitGroupSeparator);
69 return loc.toString(value, 'g', 18);
70}
71
72/** \brief convert a double to a string
73 * \ingroup jkqtptools_string
74 */
75inline QString JKQTPDoubleToQString(double value, int prec = 10, char f = 'g', QChar decimalSeparator='.') {
76 QLocale loc=QLocale::c();
77 loc.setNumberOptions(QLocale::OmitGroupSeparator);
78 QString res=loc.toString(value, f, prec);
79 if (loc.decimalPoint()!=decimalSeparator) {
80 res=res.replace(loc.decimalPoint(), decimalSeparator);
81 }
82 return res;
83}
