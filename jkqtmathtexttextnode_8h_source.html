35/** \brief base class for nodes representing text in the syntax tree
36 * \ingroup jkqtmathtext_items
37 *
38 * This node is a collection of tools, necessary to draw text. It
39 * is the base for nodes, such as:
40 * - JKQTMathTextTextNode
41 * - JKQTMathTextVerbatimNode
42 * .
43 */
44class JKQTMATHTEXT_LIB_EXPORT JKQTMathTextTextBaseNode: public JKQTMathTextNode {
45 public:
46 explicit JKQTMathTextTextBaseNode(JKQTMathText* parent, const QString& text);
47 virtual ~JKQTMathTextTextBaseNode() override;
48 /** \copydoc JKQTMathTextNode::toHtml() */
49 virtual bool toHtml(QString& html, JKQTMathTextEnvironment currentEv, JKQTMathTextEnvironment defaultEv) override;
50 /** \copydoc text */
51 QString getText() const;
52 protected:
53 /** \brief text-contents of the node */
54 QString text;
55 /** \brief transforms the \a text before sizing/drawing (may e.g. exchange special letters for other unicode symbols etc.) */
56 virtual QString textTransform(const QString& text, const JKQTMathTextEnvironment& currentEv) const;
57};

106/** \brief subclass representing a verbatim (plain-text) node with support for line-breaks in the syntax tree
107 * \ingroup jkqtmathtext_items
108 *
109 * The layout of the lines can left-aligned, right-aligned or centered.
110 *
111 * \image html jkqtmathtext_verticallist.png
112 *
113 * \image html jkqtmathtext_verticalalignment.png
114 *
115 * \image html jkqtmathtext_horizontalalignment.png
116 */
117class JKQTMATHTEXT_LIB_EXPORT JKQTMathTextVerbatimNode: public JKQTMathTextTextBaseNode {
118 public:
119 explicit JKQTMathTextVerbatimNode(JKQTMathText* parent, const QString& text, bool visibleWhitespace=false, JKQTMathTextHorizontalAlignment _alignment=MTHALeft, double _linespacingFactor=1.0, JKQTMathTextVerticalOrientation _verticalOrientation=MTVOFirstLine);
120 /** \copydoc JKQTMathTextNode::getTypeName() */
121 virtual QString getTypeName() const override;
122 /** \copydoc alignment */
123 JKQTMathTextHorizontalAlignment getAlignment() const;
124 /** \copydoc verticalOrientation */
125 JKQTMathTextVerticalOrientation getVerticalOrientation() const;
126 /** \copydoc lineSpacingFactor */
127 double getLineSpacingFactor() const;
128 /** \copydoc visibleWhitespace */
129 bool getVisibleWhitespace() const;
130 /** \copydoc JKQTMathTextNode::draw() */
131 virtual double draw(QPainter& painter, double x, double y, JKQTMathTextEnvironment currentEv, const JKQTMathTextNodeSize* prevNodeSize=nullptr) override;
132 /** \copydoc JKQTMathTextNode::toHtml() */
133 virtual bool toHtml(QString& html, JKQTMathTextEnvironment currentEv, JKQTMathTextEnvironment defaultEv) override;
134 protected:
135 /** \brief alignment scheme used to lay out all lines
136 *
137 * \image html jkqtmathtext_horizontalalignment.png
138 */
139 JKQTMathTextHorizontalAlignment alignment;
140 /** \brief spacing of the separate lines, as factor of the default line-spacing [Default: 1].
141 *
142 * This property can be used to move the lines closer together or farther apart.
143 *
144 * \image html jkqtmathtext_verticallist.png
145 */
146 double lineSpacingFactor;
147 /** \brief vertical orientation of the baseline of the whole block (with respect to the single lines)
148 *
149 * \image html jkqtmathtext_verticalorientation.png
150 */
151 JKQTMathTextVerticalOrientation verticalOrientation;
152 /** \brief when \c true, whitespaces are displayed with a visible character */
153 bool visibleWhitespace;
154
155 /** \copydoc JKQTMathTextNode::getSizeInternal() */
156 virtual void getSizeInternal(QPainter& painter, JKQTMathTextEnvironment currentEv, double& width, double& baselineHeight, double& overallHeight, double& strikeoutPos, const JKQTMathTextNodeSize* prevNodeSize=nullptr) override;
157 /** \brief sets all necessary settings in \a currentEv for drawing this node */
158 virtual void transformEnvironment(JKQTMathTextEnvironment& currentEv) const;
159
160 /** \brief describes the layout of the whole node */
161 struct LayoutInfo: public JKQTMathTextNodeSize {
162 LayoutInfo();
163 /** \brief the text from JKQTMathTextVerbatimNode::text, split into lines */
164 QStringList lines;
165 /** \brief drawing position for each line */
166 QList<QPointF> X;
167 };
168 /** \brief calclates the layout of the whole block/node
169 *
170 * \note This function does NOT call transformEnvironment();
171 * it has to be called before calling this!
172 */
173 LayoutInfo calcLayout(QPainter& painter, const JKQTMathTextEnvironment& currentEv) const;
174 /** \brief transforms the \a text before sizing/drawing (may e.g. exchange special letters for other unicode symbols etc.) */
175 virtual QString textTransform(const QString& text, const JKQTMathTextEnvironment& currentEv) const override;
176};
