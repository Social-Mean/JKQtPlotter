34/** \brief subclass representing one whitepsace node in the syntax tree
35 * \ingroup jkqtmathtext_items
36 *
37 * It can represent different types of spaces, see Types and different multiples of that space.
38 * so \c JKQTMathTextWhitespaceNode(WST1em,2,parent) will represent two 1em wide spaces, i.e. a 2em space.
39 */
40class JKQTMATHTEXT_LIB_EXPORT JKQTMathTextWhitespaceNode: public JKQTMathTextNode {
41 public:
42 enum Types {
43 WSTNormal, /*!< \brief a normal-width whitespace */
44 WSTNonbreaking, /*!< \brief a normal-width, non-breaking whitespace */
45 WST1en, /*!< \brief 1en (= 0.5 * pointsize of the font) whitespace */
46 WST1em, /*!< \brief 1em (= pointsize of the font) whitespace */
47 WSTQuad=WST1em, /*!< \copydoc WST1em */
48 WSThair, /*!< \brief hair (thinner than thin) whitespace (1/12 em = pointsize/12) */
49 WSTthin, /*!< \brief thin whitespace (1/6 em = pointsize/6) */
50 WSTnegthin, /*!< \brief negative thin whitespace (-1/6 em = -pointsize/6) */
51 WSTmedium, /*!< \brief medium whitespace (2/9 em = pointsize*2/9) */
52 WSTnegmedium, /*!< \brief negative medium whitespace (-2/9 em = -pointsize*2/9) */
53 WSTthick, /*!< \brief thick whitespace (5/18 em = pointsize*5/18) */
54 WSTnegthick, /*!< \brief negative thick whitespace (-5/18 em = -pointsize*5/18) */
55 WSTthicker, /*!< \brief thicker whitespace (1/3 em = pointsize/3) */
56
57 };
